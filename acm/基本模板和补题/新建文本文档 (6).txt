#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<set>
#define ll long long
using namespace std;
const int inf=1e9+7;
const int MAXN=1e5+5;
struct node{
    int ind;
    int x;
}a[MAXN];
bool cmp(node a,node b){
    if(a.x!=b.x)  return a.x>b.x;
    return a.ind<b.ind;
}
int b[MAXN];
int main(){
    int T,n;
    scanf("%d",&T);
    while(T--){
        memset(b,0,sizeof(b));
        scanf("%d",&n);
        for(int i=1; i<=n; i++){
            scanf("%d",&a[i].x);
            a[i].ind=i;
        }
        sort(a+1,a+n+1,cmp);
        node temp={inf,-1};
        for(int i=1;i<=n;i++){
            if(temp.x!=a[i].x){
                temp.x=a[i].x;
                int tt=sqrt(a[i].ind);
                for(int j=2;j<=tt;j++){
                    if(a[i].ind%j!=0){
                        b[j]=max(b[j],a[i].x);
                        printf("111\n");
                    }
                }
                if((i+1<=n)&&a[i].ind<temp.ind){
                        b[i+1]=a[i].x;
                temp.ind=min(temp.ind,i+1);
                printf("2222\n");
                }
            }
        }
        int t1=b[2];
        for(int i=0;i<=10;i++)
            printf("%d ",b[i]);
        printf("\n");
        for(int i=2; i<=n; i++){
            if(b[i]==0)  printf("%d",t1);
            else{
                t1=b[i];
                if(i!=n)
                    printf("%d ",b[i]);
                else printf("%d",b[i]);
            }
        }
        printf("\n");
    }
    return 0;
}
